iteracion 0 Datos Generales
--------------------------------------------------------------------------------------
Academik es una aplicación web responsive orientada al control de asistencia escolar.
Permitirá que varios profesores accedan a la lista de alumnos de un curso y sección, 
registren las ausencias durante el llamado y generen reportes de asistencia para 
seguimiento académico y administrativo.

Objetivo general:
Desarrollar una aplicación web responsive para gestionar la asistencia escolar de 
forma simple, rápida y segura, permitiendo registrar ausencias por curso y sección y generar reportes posteriores.

Objetivos específicos:
1. Permitir que los profesores accedan a sus listas de alumnos.
2. Registrar ausencias durante el llamado.
3. Generar reportes de asistencia.
4. Importar alumnos, cursos y secciones desde archivos PDF.
5. Administrar manualmente la información mediante pantallas ABM.
6. Mantener una base modular y preparada para crecer.

Enfoque de uso:
La aplicación estará pensada principalmente para celulares, aunque también podrá usarse desde computadoras.
La interfaz debe priorizar rapidez, claridad y facilidad de uso para docentes.

Criterios técnicos:
1. Arquitectura modular.
2. Código mantenible.
3. Diseño responsive enfocada a dispositivos mobiles.
4. Facilidad de despliegue.
5. Preparado para escalabilidad futura.

CREATE TABLE roles (
    id              BIGSERIAL PRIMARY KEY,
    nombre          VARCHAR(50) NOT NULL UNIQUE,
    descripcion     VARCHAR(255),
    activo          BOOLEAN NOT NULL DEFAULT TRUE,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE usuarios (
    id              BIGSERIAL PRIMARY KEY,
    username        VARCHAR(150) NOT NULL UNIQUE,
    password_hash   VARCHAR(255) NOT NULL,
    email           VARCHAR(255),
    rol_id          BIGINT NOT NULL REFERENCES roles(id) ON DELETE RESTRICT,
    activo          BOOLEAN NOT NULL DEFAULT TRUE,
    ultimo_login    TIMESTAMPTZ,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE profesores (
    id              BIGSERIAL PRIMARY KEY,
    usuario_id      BIGINT UNIQUE REFERENCES usuarios(id) ON DELETE SET NULL,
    cedula          VARCHAR(30) NOT NULL UNIQUE,
    nombres         VARCHAR(100) NOT NULL,
    apellidos       VARCHAR(100) NOT NULL,
    telefono        VARCHAR(30),
    email           VARCHAR(255),
    activo          BOOLEAN NOT NULL DEFAULT TRUE,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE cursos (
    id                  BIGSERIAL PRIMARY KEY,
    codigo              VARCHAR(30) NOT NULL,
    nombre              VARCHAR(150) NOT NULL,
    seccion             VARCHAR(30) NOT NULL,
    turno               VARCHAR(30) NOT NULL,
    periodo_lectivo     VARCHAR(30) NOT NULL,
    nivel               VARCHAR(50),
    grado               VARCHAR(50),
    activo              BOOLEAN NOT NULL DEFAULT TRUE,
    creado              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT uq_cursos_codigo_periodo_seccion_turno
        UNIQUE (codigo, periodo_lectivo, seccion, turno)
);

CREATE TABLE profesor_curso (
    id              BIGSERIAL PRIMARY KEY,
    profesor_id     BIGINT NOT NULL REFERENCES profesores(id) ON DELETE CASCADE,
    curso_id        BIGINT NOT NULL REFERENCES cursos(id) ON DELETE CASCADE,
    titular         BOOLEAN NOT NULL DEFAULT FALSE,
    activo          BOOLEAN NOT NULL DEFAULT TRUE,
    fecha_desde     DATE NOT NULL,
    fecha_hasta     DATE,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_profesor_curso_fechas
        CHECK (fecha_hasta IS NULL OR fecha_hasta >= fecha_desde),
    CONSTRAINT uq_profesor_curso
        UNIQUE (profesor_id, curso_id, fecha_desde)
);

CREATE TABLE alumnos (
    id                      BIGSERIAL PRIMARY KEY,
    documento               VARCHAR(30) NOT NULL UNIQUE,
    nombres                 VARCHAR(100) NOT NULL,
    apellidos               VARCHAR(100) NOT NULL,
    fecha_nacimiento        DATE,
    sexo                    VARCHAR(20),
    direccion               VARCHAR(255),
    telefono_responsable    VARCHAR(30),
    observaciones           TEXT,
    activo                  BOOLEAN NOT NULL DEFAULT TRUE,
    creado                  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_alumnos_sexo
        CHECK (sexo IS NULL OR sexo IN ('M', 'F', 'OTRO'))
);

CREATE TABLE alumno_curso (
    id              BIGSERIAL PRIMARY KEY,
    alumno_id       BIGINT NOT NULL REFERENCES alumnos(id) ON DELETE CASCADE,
    curso_id        BIGINT NOT NULL REFERENCES cursos(id) ON DELETE CASCADE,
    fecha_ingreso   DATE NOT NULL,
    fecha_salida    DATE,
    estado          VARCHAR(20) NOT NULL DEFAULT 'ACTIVO',
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_alumno_curso_fechas
        CHECK (fecha_salida IS NULL OR fecha_salida >= fecha_ingreso),
    CONSTRAINT chk_alumno_curso_estado
        CHECK (estado IN ('ACTIVO', 'RETIRADO', 'TRASLADADO', 'FINALIZADO')),
    CONSTRAINT uq_alumno_curso
        UNIQUE (alumno_id, curso_id, fecha_ingreso)
);

CREATE TABLE asistencias (
    id                  BIGSERIAL PRIMARY KEY,
    curso_id            BIGINT NOT NULL REFERENCES cursos(id) ON DELETE RESTRICT,
    profesor_id         BIGINT NOT NULL REFERENCES profesores(id) ON DELETE RESTRICT,
    fecha               DATE NOT NULL,
    hora_inicio         TIME,
    hora_confirmacion   TIMESTAMPTZ,
    estado_control      VARCHAR(20) NOT NULL DEFAULT 'BORRADOR',
    observacion         TEXT,
    creado_por_id       BIGINT REFERENCES usuarios(id) ON DELETE SET NULL,
    creado              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_asistencias_estado
        CHECK (estado_control IN ('BORRADOR', 'CONTROLADA', 'CORREGIDA')),
    CONSTRAINT uq_asistencias_curso_profesor_fecha
        UNIQUE (curso_id, profesor_id, fecha)
);

CREATE TABLE asistencia_detalle (
    id              BIGSERIAL PRIMARY KEY,
    asistencia_id   BIGINT NOT NULL REFERENCES asistencias(id) ON DELETE CASCADE,
    alumno_id       BIGINT NOT NULL REFERENCES alumnos(id) ON DELETE RESTRICT,
    estado          VARCHAR(20) NOT NULL DEFAULT 'PRESENTE',
    observacion     TEXT,
    corregido       BOOLEAN NOT NULL DEFAULT FALSE,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_asistencia_detalle_estado
        CHECK (estado IN ('PRESENTE', 'AUSENTE', 'JUSTIFICADO')),
    CONSTRAINT uq_asistencia_detalle
        UNIQUE (asistencia_id, alumno_id)
);

CREATE TABLE importaciones_pdf (
    id                  BIGSERIAL PRIMARY KEY,
    nombre_archivo      VARCHAR(255) NOT NULL,
    ruta_archivo        VARCHAR(500),
    tipo_importacion    VARCHAR(50) NOT NULL,
    estado              VARCHAR(20) NOT NULL DEFAULT 'CARGADO',
    total_registros     INTEGER NOT NULL DEFAULT 0,
    total_ok            INTEGER NOT NULL DEFAULT 0,
    total_error         INTEGER NOT NULL DEFAULT 0,
    usuario_id          BIGINT REFERENCES usuarios(id) ON DELETE SET NULL,
    fecha_proceso       TIMESTAMPTZ,
    creado              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_importaciones_pdf_estado
        CHECK (estado IN ('CARGADO', 'PROCESADO', 'VALIDADO', 'IMPORTADO', 'ERROR'))
);

CREATE TABLE importacion_pdf_detalle (
    id              BIGSERIAL PRIMARY KEY,
    importacion_id  BIGINT NOT NULL REFERENCES importaciones_pdf(id) ON DELETE CASCADE,
    nro_linea       INTEGER,
    tipo_registro   VARCHAR(50),
    dato_original   TEXT,
    mensaje_error   TEXT,
    procesado_ok    BOOLEAN NOT NULL DEFAULT FALSE,
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE notificaciones (
    id                  BIGSERIAL PRIMARY KEY,
    tipo                VARCHAR(30) NOT NULL,
    destinatario        VARCHAR(255) NOT NULL,
    asunto              VARCHAR(255),
    mensaje             TEXT NOT NULL,
    estado_envio        VARCHAR(20) NOT NULL DEFAULT 'PENDIENTE',
    fecha_envio         TIMESTAMPTZ,
    referencia_entidad  VARCHAR(100),
    referencia_id       BIGINT,
    creado              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    modificado          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_notificaciones_estado
        CHECK (estado_envio IN ('PENDIENTE', 'ENVIADO', 'ERROR', 'CANCELADO'))
);

CREATE TABLE auditoria (
    id              BIGSERIAL PRIMARY KEY,
    usuario_id      BIGINT REFERENCES usuarios(id) ON DELETE SET NULL,
    entidad         VARCHAR(100) NOT NULL,
    entidad_id      BIGINT,
    accion          VARCHAR(50) NOT NULL,
    valor_anterior  JSONB,
    valor_nuevo     JSONB,
    ip              VARCHAR(50),
    user_agent      TEXT,
    fecha_hora      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    creado          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE OR REPLACE FUNCTION fn_actualizar_modificado()
RETURNS TRIGGER AS $$
BEGIN
    NEW.modificado = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

Dinamic Island
Entendí que no querés una curva superior tipo “mordida” o cerrándose hacia adentro, sino una silueta donde el borde superior salga hacia los laterales, más parecido al remate exterior del módulo superior en iPhone modernos. En otras palabras: mantener el DI pegado arriba, pero cambiar la forma de las esquinas superiores para que abran hacia los costados en vez de redondearse hacia dentro.
Voy a tocar solo la geometría del DI. La idea es mantener el cuerpo central, pero agregar “hombros” laterales arriba para que la curva superior abra hacia los costados en vez de cerrarse hacia adentro.
